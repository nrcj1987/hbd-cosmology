from setuptools import setup, find_packages

with open("README.md", "r", encoding="utf-8") as fh:
    long_description = fh.read()

with open("requirements.txt", "r", encoding="utf-8") as fh:
    requirements = [line.strip() for line in fh if line.strip() and not line.startswith("#")]

setup(
    name="hbd-cosmology",
    version="1.0.0",
    author="Nicholas R. C. Jimison",
    author_email="nrcj1987@gmail.com",
    description="Hyperbolic Brane Dynamics cosmological model implementation",
    long_description=long_description,
    long_description_content_type="text/markdown",
    url="https://github.com/nrcj1987/hbd-cosmology",
    package_dir={"": "src"},
    packages=find_packages(where="src"),
    classifiers=[
        "Development Status :: 4 - Beta",
        "Intended Audience :: Science/Research",
        "License :: OSI Approved :: MIT License",
        "Operating System :: OS Independent",
        "Programming Language :: Python :: 3",
        "Programming Language :: Python :: 3.8",
        "Programming Language :: Python :: 3.9",
        "Programming Language :: Python :: 3.10",
        "Topic :: Scientific/Engineering :: Astronomy",
        "Topic :: Scientific/Engineering :: Physics",
    ],
    python_requires=">=3.8",
    install_requires=requirements,
    entry_points={
        "console_scripts": [
            "hbd-run=scripts.run_hbd:main",
            "hbd-plot=scripts.plot_spectra:main",
        ],
    },
)
